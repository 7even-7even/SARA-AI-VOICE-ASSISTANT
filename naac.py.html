import pyttsx3
import os
import time

def welcome_message():
    engine = pyttsx3.init()

    # Change the speech rate (speed)
    rate = engine.getProperty('rate')  # Get current speed
    engine.setProperty('rate', 140)  # Set new speed (lower = slower, higher = faster)

    # Change the volume (optional)
    volume = engine.getProperty('volume')  # Get current volume level (0.0 to 1.0)
    engine.setProperty('volume', 1.0)  # Set volume level (0.0 to 1.0)

    engine.say("Welcome! Honorable National Assessment and Accreditation Council Peer Team.")
    engine.runAndWait()
